• forEach 与 for 区别性特点

    forEach()返回值是undefined，不可以链式调用

    但是for循环可以

    封装获取查询字符串的值

    写法一:
getSearch(x) {
    var info = window.location.search;
    var arrs = info.slice(1,info.length).split('&');
    var y;
    arrs.forEach((v)=>{
        var arrAy = v.split('&');
        var objarr = arrAy[0].split('=')
        if(objarr[0] == x){
            y = objarr[1];
        }
    });
    return y;
}
写法二:
getSearch(x) {
    var info = window.location.search;
    var arrs = info.slice(1,info.length).split('&');
    for(var i=0;i <= arrs.length;i++){
         var arrAy = arrs[i].split('&');
         var objarr = arrAy[0].split('=')
         return objarr[1];
    }
}
